\poemtitle{Harlem}
\settowidth{\versewidth}{like a raisin in the sun?}

\newcommand{\attrib}[1]{%

\begin{verse}[\versewidth]

What happens to a dream deferred?

\vin Does it dry up \\
\vin like a raisin in the sun? \\
\vin Or fester like a sore--- \\
\vin And then run? \\
\vin Does it stink like rotten meat? \\
\vin Or crust and sugar over--- \\
\vin like a syrupy sweet?

\vin Maybe it just sags \\
\vin like a heavy load.

\vin \textit{Or does it explode?}

\end{verse}

\attrib{Langston Hughes (1901-1967)}
